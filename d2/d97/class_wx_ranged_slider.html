void WxRangedSlider::OnMouseEvent ( wxMouseEvent &  InEvent)

Mouse message for the RangedSlider

WPF Implementation INT DeltaX = InEvent.GetX() - LastMousePoint.x; LastMousePoint = InEvent.GetPosition();

if (DeltaX != 0) { RangedNumeric.MoveByIncrements (DeltaX*Abs(DeltaX)); PropertyInput->SetValue( RangedNumeric.GetStringValue(), PropertyItem ); PropertyItem->Refresh(); PropertyItem->Update(); }
